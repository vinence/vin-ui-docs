# Dialog 服务

用于动态创建对话框的服务函数。

# 安装

import Vue from 'vue';
import { dialogService } from 'vin-ui';

Vue.prototype.$dialog = dialogService({
  store,   // Vuex store 实例
  router,  // Vue Router 实例
  Vue      // Vue 构造函数
});

# 基础用法

通过 dialogService 可以快速创建一个对话框。

<template>
  <div class="example-dialog-service">
    <el-button type="primary" @click="openBasicDialog">
      基础对话框
    </el-button>
    <el-button type="success" @click="openContentDialog">
      自定义内容
    </el-button>
    <el-button type="warning" @click="openPromiseDialog">
      Promise 支持
    </el-button>
  </div>
</template>

<script>
export default {
  methods: {
    // 基础对话框
    openBasicDialog() {
      this.$dialog({
        dialogProps: {
          title: '基础对话框',
          width: '30%'
        },
        content: '这是一个基础的对话框示例'
      });
    },
    // 自定义内容对话框
    openContentDialog() {
      this.$dialog({
        dialogProps: {
          title: '自定义内容',
          width: '400px'
        },
        content: `
          <div style="text-align: center;">
            <i class="el-icon-success" style="font-size: 48px; color: #67C23A;"></i>
            <p style="margin: 20px 0;">操作成功！</p>
            <p style="color: #909399; font-size: 14px;">
              这是一个带有自定义内容的对话框示例
            </p>
          </div>
        `
      });
    },
    // Promise 支持示例
    openPromiseDialog() {
      this.$dialog({
        dialogProps: {
          title: 'Promise 支持',
          width: '30%'
        },
        content: '确认执行此操作吗？',
        footerBtn: [
          {
            label: '取消',
            type: 'default'
          },
          {
            label: '确定',
            type: 'primary'
          }
        ]
      })
      .then(() => {
        this.$message.success('操作已确认');
      })
      .catch(() => {
        this.$message.info('操作已取消');
      });
    }
  }
}
</script>

  margin-right: 15px;
}

# 异步组件

支持加载异步组件作为对话框内容。

<template>
  <div class="example-dialog-service">
    <el-button type="primary" @click="openAsyncDialog">
      异步组件对话框
    </el-button>
    <el-button type="success" @click="openFormDialog">
      表单对话框
    </el-button>
  </div>
</template>

<script>
// 异步加载的表单组件
const AsyncForm = {
  template: `
    <el-form :model="form" label-width="80px">
      <el-form-item label="姓名">
        <el-input v-model="form.name"></el-input>
      </el-form-item>
      <el-form-item label="年龄">
        <el-input-number v-model="form.age" :min="1"></el-input-number>
      </el-form-item>
      <el-form-item label="地址">
        <el-input v-model="form.address" type="textarea"></el-input>
      </el-form-item>
    </el-form>
  `,
  data() {
    return {
      form: {
        name: '',
        age: 18,
        address: ''
      }
    }
  }
}

export default {
  methods: {
    // 异步组件示例
    openAsyncDialog() {
      const AsyncComponent = () => ({
        component: new Promise(resolve => {
          setTimeout(() => {
            resolve({
              template: `
                <div class="async-content">
                  <div class="loading-placeholder">
                    <i class="el-icon-loading"></i>
                    <p>异步加载的内容</p>
                  </div>
                </div>
              `,
              data() {
                return {
                  message: '这是异步加载的组件内容'
                }
              }
            })
          }, 1000)
        }),
        loading: {
          template: `
            <div style="text-align: center; padding: 20px;">
              <i class="el-icon-loading"></i>
              <p>加载中...</p>
            </div>
          `
        }
      })

      this.$dialog({
        dialogProps: {
          title: '异步组件',
          width: '500px'
        },
        asyncComp: AsyncComponent
      });
    },
    // 表单对话框示例
    openFormDialog() {
      this.$dialog({
        dialogProps: {
          title: '用户信息',
          width: '500px'
        },
        asyncComp: AsyncForm,
        footerBtn: [
          {
            label: '取消',
            type: 'default'
          },
          {
            label: '提交',
            type: 'primary',
            handler: (instance) => {
              console.log('表单数据:', instance.$children[0].form);
              instance.close();
            }
          }
        ]
      });
    }
  }
}
</script>

# 自定义按钮

可以自定义底部按钮的配置。

<template>
  <div class="example-dialog-service">
    <el-button type="primary" @click="openCustomDialog">
      打开自定义按钮对话框
    </el-button>
  </div>
</template>

<script>
export default {
  methods: {
    openCustomDialog() {
      this.$dialog({
        dialogProps: {
          title: '自定义按钮对话框',
          width: '30%'
        },
        content: '这个对话框有自定义的底部按钮',
        footerBtn: [
          {
            label: '保存草稿',
            type: 'info',
            if: true,
            handler: () => {
              console.log('保存草稿');
            }
          },
          {
            label: '提交',
            type: 'primary',
            if: true,
            handler: () => {
              console.log('提交');
            }
          }
        ]
      });
    }
  }
}
</script>

# Promise 支持

dialogService 返回一个 Promise，可以通过 then/catch 处理对话框的确认和取消事件。

this.$dialog({
  title: '提示',
  content: '确认执行此操作？'
})
  .then(res => {
    console.log('确认', res);
  })
  .catch(err => {
    console.log('取消', err);
  });

# API

# DialogProps

参数 说明 类型 默认值
title 对话框标题 string ''
width 对话框宽度 string '50%'
height 对话框高度 string ''
top 对话框距顶部距离 string '8vh'
showClose 是否显示关闭按钮 boolean true
modal 是否显示遮罩层 boolean true
closeOnClickModal 是否点击遮罩层关闭 boolean false
closeOnPressEscape 是否按ESC关闭 boolean false
fullscreen 是否全屏显示 boolean false
center 是否居中显示 boolean true
class 自定义类名 string ''
customClass 自定义额外类名 string ''
lockScroll 是否锁定滚动 boolean true
modalAppendToBody 遮罩层是否插入至body元素上 boolean true

# DialogOptions

参数 说明 类型 默认值
asyncComp 异步组件 Object -
dialogProps 对话框属性配置 DialogProps -
compProps 组件属性 Object -
footerBtn 底部按钮配置 FooterButton[] -
content 对话框内容 string/VNode -

# FooterButton

参数 说明 类型 默认值
if 是否显示按钮的条件 boolean true
text 按钮文本 string -
type 按钮类型 'primary'/'success'/'warning'/'danger'/'info' -
handler 点击回调函数 Function -
loading 是否显示加载状态 boolean false
disabled 是否禁用状态 boolean false

# 方法

方法名 说明 参数 返回值
close 关闭对话框 - -
destroy 销毁对话框实例 - -
setLoading 设置按钮加载状态 (index: number, loading: boolean) -
setButtonDisabled 设置按钮禁用状态 (index: number, disabled: boolean) -

# 事件

事件名 说明 回调参数
open 对话框打开时触发 -
close 对话框关闭时触发 -
closed 对话框关闭动画结束时触发 -
