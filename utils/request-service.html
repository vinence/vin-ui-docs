# Request 服务

一个功能强大的 HTTP 请求库，支持请求队列管理、缓存、加密、页面级请求控制等特性。

# 安装

npm install vin-ui
1

# 基础配置

import { createRequest } from 'vin-ui';

// 创建请求实例
const request = createRequest({
  // 基础配置
  baseURL: 'https://api.example.com',
  timeout: 10000,
  withCredentials: true,
  
  // 请求头配置
  headers: {
    common: {
      'Content-Type': 'application/json;charset=UTF-8',
      'Accept': 'application/json, text/plain, */*'
    },
    // 自定义静态请求头
    custom: {
      'X-Custom-Header': 'custom-value'
    }
  },

  // Loading 配置
  loading: {
    enabled: true,
    service: {
      show: () => ElLoading.service({ fullscreen: true }),
      hide: (instance) => instance.close()
    },
    delay: 300,        // loading 显示延迟(ms)
    minDuration: 500   // 最小显示时间(ms)
  },

  // 缓存配置
  cache: {
    enabled: true,
    // 可选的缓存类型: temporary(临时), memory(内存), session(会话), persistent(持久)
    type: 'memory',
    maxAge: 60000,  // 缓存有效期(ms)
    // 排除的请求路径
    exclude: [/\/(create|update|delete)/, /\/realtime/],
    // 排除的请求方法
    excludeMethods: ['PUT', 'DELETE']
  },

  // 错误处理配置
  error: {
    // 全局错误处理器
    handler: (error) => {
      const { response } = error;
      if (response?.status === 401) {
        // 处理未授权
        router.push('/login');
      } else {
        ElMessage.error(error.message || '请求失败');
      }
    },
    // 错误重试配置
    retry: {
      enabled: true,
      maxRetries: 3,
      delay: 1000,
      retryableStatusCodes: [408, 500, 502, 503, 504]
    }
  },

  // 认证配置
  auth: {
    token: {
      enabled: true,
      headerKey: 'Authorization',
      prefix: 'Bearer ',
      storage: {
        type: 'localStorage',
        key: 'token'
      }
    }
  }
});

// 全局挂载
Vue.prototype.$request = request;

# 基础用法

发起基本的 GET/POST 请求。

<template>
  <div class="example-request-service">
    <div class="button-group">
      <el-button type="primary" @click="getData" :loading="loading.get">
        GET请求
      </el-button>
      <el-button type="success" @click="postData" :loading="loading.post">
        POST请求
      </el-button>
      <el-button type="warning" @click="getWithEncrypt" :loading="loading.encrypt">
        加密请求
      </el-button>
    </div>

    <div v-if="responseData" class="response-data">
      <div class="response-header">
        <span>响应数据:</span>
        <el-tag size="small" :type="responseData.__fromCache ? 'info' : 'success'">
          {{ responseData.__fromCache ? '来自缓存' : '来自服务器' }}
        </el-tag>
      </div>
      <div class="response-content">
        <pre>{{ JSON.stringify(responseData.data, null, 2) }}</pre>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: {
        get: false,
        post: false,
        encrypt: false
      },
      responseData: null
    }
  },

  methods: {
    async getData() {
      this.loading.get = true
      try {
        const data = await this.$request.get({
          url: '/api/users',
          data: { page: 1, pageSize: 5 }
        })
        this.responseData = data
        this.$message.success('GET请求成功')
      } catch (error) {
        console.error('GET请求失败:', error)
        this.$message.error(error.message || 'GET请求失败')
      } finally {
        this.loading.get = false
      }
    },

    async postData() {
      this.loading.post = true
      try {
        const data = await this.$request.post({
          url: '/api/users',
          data: {
            name: '张三',
            age: 25
          }
        })
        this.responseData = data
        this.$message.success('POST请求成功')
      } catch (error) {
        console.error('POST请求失败:', error)
        this.$message.error(error.message || 'POST请求失败')
      } finally {
        this.loading.post = false
      }
    },

    async getWithEncrypt() {
      this.loading.encrypt = true
      try {
        const data = await this.$request.get({
          url: '/api/users',
          data: { page: 1, pageSize: 5 },
          flags: {
            enableEncrypt: true
          }
        })
        this.responseData = data
        this.$message.success('加密请求成功')
      } catch (error) {
        console.error('加密请求失败:', error)
        this.$message.error(error.message || '加密请求失败')
      } finally {
        this.loading.encrypt = false
      }
    }
  }
}
</script>

# 请求队列管理

支持页面级请求管理，自动处理 loading 状态和请求取消。

<template>
  <div class="example-request-queue">
    <div class="button-group">
      <el-button type="primary" @click="loadPageData" :loading="loading">
        并发加载数据
      </el-button>
      <el-button type="danger" @click="cancelRequests" :disabled="!loading">
        取消所有请求
      </el-button>
    </div>

    <div v-if="responseData.length" class="response-data">
      <div class="response-header">
        <span>响应数据:</span>
        <el-tag size="small" type="success">{{ responseData.length }}个请求</el-tag>
      </div>
      <div class="data-list">
        <div v-for="(item, index) in responseData" :key="index" class="data-item">
          <div class="data-title">请求 {{index + 1}}</div>
          <div class="response-content">
            <pre>{{ JSON.stringify(item.data, null, 2) }}</pre>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      responseData: []
    }
  },

  methods: {
    async loadPageData() {
      this.loading = true
      this.responseData = []

      try {
        // 并发请求,共享loading状态
        const results = await Promise.all([
          this.$request.get({
            url: '/api/users',
            data: { type: 'users' }
          }),
          this.$request.get({
            url: '/api/posts',
            data: { type: 'posts' }
          }),
          this.$request.get({
            url: '/api/comments',
            data: { type: 'comments' }
          })
        ])

        this.responseData = results
        this.$message.success('数据加载成功')
      } catch (error) {
        this.$message.error('加载失败')
        console.error('加载失败:', error)
      }

      this.loading = false
    },

    cancelRequests() {
      // 取消所有进行中的请求
      this.$request.cancelAll()
      this.loading = false
      this.$message.info('已取消所有请求')
    }
  },

  beforeDestroy() {
    // 组件销毁时取消请求
    this.cancelRequests()
  }
}
</script>

# 缓存请求

支持多种缓存策略：临时缓存、内存缓存、会话缓存和持久化缓存。

<template>
  <div class="example-request-cache">
    <div class="button-group">
      <el-button-group>
        <el-button 
          v-for="type in cacheTypes"
          :key="type.value"
          :type="currentType === type.value ? 'primary' : ''"
          @click="setCacheType(type.value)">
          {{type.label}}
        </el-button>
      </el-button-group>

      <el-button type="warning" @click="clearCache">
        清除缓存
      </el-button>
    </div>

    <div class="operation-group">
      <el-button type="success" @click="loadData" :loading="loading">
        发起请求
      </el-button>
      <div class="cache-info" v-if="requestCount > 0">
        <div class="info-item">
          <span class="label">请求次数:</span>
          <el-tag size="small">{{requestCount}}</el-tag>
        </div>
        <div class="info-item">
          <span class="label">缓存命中:</span>
          <el-tag size="small" type="success">{{cacheHitCount}}</el-tag>
        </div>
      </div>
    </div>

    <div v-if="responseData" class="response-data">
      <div class="response-header">
        <span>响应数据:</span>
        <el-tag size="small" :type="responseData.__CACHE_HIT__ ? 'warning' : 'success'">
          {{ responseData.__CACHE_HIT__ ? '来自缓存' : '来自服务器' }}
        </el-tag>
        <span class="cache-time" v-if="responseData.__CACHE_HIT__">
          缓存时间: {{ formatTime(responseData.__timestamp) }}
        </span>
      </div>
      <div class="response-content">
        <pre>{{ JSON.stringify(responseData.data, null, 2) }}</pre>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      responseData: null,
      currentType: 'none',
      requestCount: 0,
      cacheHitCount: 0,
      cacheTypes: [
        { label: '无缓存', value: 'none' },
        { label: '临时缓存', value: 'temporary' },
        { label: '内存缓存', value: 'memory' },
        { label: '会话缓存', value: 'session' },
        { label: '持久缓存', value: 'persistent' }
      ]
    }
  },

  methods: {
    setCacheType(type) {
      this.currentType = type
      this.responseData = null
      this.requestCount = 0
      this.cacheHitCount = 0
    },

    async loadData() {
      this.loading = true
      this.requestCount++

      try {
        const data = await this.$request.get({
          url: '/api/users',
          data: { id: 1 }
        }, {
          enableCache: this.currentType !== 'none',
          cacheType: this.currentType
        })
        if (data.__CACHE_HIT__) {
          this.cacheHitCount++
        }

        this.responseData = data
        this.$message.success(data.__CACHE_HIT__ ? '从缓存获取成功' : '从服务器获取成功')
      } catch (error) {
        this.$message.error('请求失败')
        console.error('请求失败:', error)
      }

      this.loading = false
    },

    clearCache() {
      this.$request.clearCache()
      this.responseData = null
      this.$message.success('缓存已清除')
    },

    formatTime(timestamp) {
      if (!timestamp) return ''
      const date = new Date(timestamp)
      return date.toLocaleTimeString()
    }
  }
}
</script>

# 轮询请求

支持轮询请求，可配置轮询间隔和条件。

轮询间隔:
ms
<template>
  <div class="example-request-polling">
    <div class="button-group">
      <el-button type="primary" @click="startPolling" :disabled="polling">
        开始轮询
      </el-button>
      <el-button type="danger" @click="handleStopPolling" :disabled="!polling">
        停止轮询
      </el-button>
    </div>

    <div class="config-panel">
      <div class="config-item">
        <span class="label">轮询间隔:</span>
        <el-input-number 
          v-model="interval" 
          :min="1000"
          :step="1000"
          :disabled="polling"
          size="small">
        </el-input-number>
        <span class="unit">ms</span>
      </div>
      <div class="config-item">
        <el-checkbox 
          v-model="stopOnError"
          :disabled="polling">
          出错时停止
        </el-checkbox>
      </div>
    </div>

    <div class="polling-info" v-if="pollingData.length">
      <div class="info-header">
        <span>轮询数据:</span>
        <el-tag size="small" type="success">
          已轮询 {{pollCount}} 次
        </el-tag>
      </div>
      <div class="data-list">
        <div 
          v-for="(item, index) in pollingData"
          :key="index"
          class="data-item">
          <div class="data-time">{{formatTime(item.timestamp)}}</div>
          <div class="data-content">{{item.status}}</div>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      polling: false,
      stopPolling: null,
      pollingData: [],
      pollCount: 0,
      interval: 3000,
      stopOnError: true,
      maxItems: 5
    }
  },

  methods: {
    startPolling() {
      this.polling = true
      this.pollCount = 0
      this.pollingData = []

      // 开始轮询
      this.stopPolling = this.$request.poll({
        url: '/api/status'
      }, {
        interval: this.interval,
        immediate: true,
        stopOnError: this.stopOnError,
        maxRetries: 0,
        onSuccess: (data) => {
          this.pollCount++
          this.pollingData.unshift({
            status: data.data.status,
            timestamp: data.data.timestamp
          })
          // 只保留最新的N条数据
          if (this.pollingData.length > this.maxItems) {
            this.pollingData.pop()
          }
        },
        onError: (error) => {
          this.$message.error('轮询请求失败')
          console.error('轮询失败:', error)
          if (this.stopOnError) {
            this.polling = false
          }
        }
      })
    },

    handleStopPolling() {
      if (this.stopPolling) {
        this.stopPolling()
        this.polling = false
        this.$message.info('轮询已停止')
      }
    },

    formatTime(timestamp) {
      if (!timestamp) return ''
      const date = new Date(timestamp)
      return date.toLocaleTimeString()
    }
  },

  beforeDestroy() {
    this.stopPolling?.()
  }
}
</script>

# 错误处理

请求服务支持全局错误处理和请求级别的错误处理，并提供灵活的重试机制。

<template>
  <div class="example-request-error">
    <el-button type="primary" @click="handleGlobalError">
      触发全局错误
    </el-button>
    <el-button type="success" @click="handleRequestError">
      触发请求级错误
    </el-button>
    <el-button type="warning" @click="handleRetryError">
      触发重试错误
    </el-button>
  </div>
</template>

<script>
export default {
  methods: {
    // 触发全局错误
    async handleGlobalError() {
      try {
        await this.$request.get({
          url: '/api/error/global'
        });
      } catch (error) {
        console.log('error ---------------', error);
        console.log('全局错误已处理');
      }
    },

    // 触发请求级错误
    async handleRequestError() {
      try {
        await this.$request.get({
          url: '/api/error/request'
        }, {
          enableLoading: true
        }, {
          error: {
            handler: (error) => {
              this.$message.warning('请求级错误：' + error.message);
              return Promise.reject(error);
            }
          }
        });
      } catch (error) {
        console.log('请求级错误已处理');
      }
    },

    // 触发重试错误
    async handleRetryError() {
      try {
        await this.$request.get({
          url: '/api/error/retry'
        }, {
          enableLoading: true,
          enableRetry: true
        }, {
          error: {
            retry: {
              maxRetries: 2,
              retryDelay: 1000
            }
          }
        });
      } catch (error) {
        this.$message.error('重试后仍然失败');
      }
    }
  }
};
</script>

# 错误处理说明

  1. 全局错误处理

    • 统一处理常见的错误状态码（401、403、500等）
    • 自动显示错误提示
    • 支持错误拦截和转换

  2. 请求级错误处理

    • 可以针对特定请求自定义错误处理逻辑
    • 优先级高于全局错误处理
    • 支持错误信息自定义

  3. 重试机制

    • 支持配置重试次数和延迟
    • 可以指定需要重试的状态码
    • 重试完成后仍失败则抛出错误

# 配置说明

// 全局错误处理配置
const request = createRequest({
  error: {
    // 全局错误处理器
    handler: (error) => {
      const { response } = error;
      if (response?.status === 401) {
        router.push('/login');
      } else {
        ElMessage.error(error.message || '请求失败');
      }
    },
    // 错误重试配置
    retry: {
      enabled: true,
      maxRetries: 3,
      delay: 1000,
      retryableStatusCodes: [408, 500, 502, 503, 504]
    }
  }
});

// 请求级错误处理
request.get('/api/data', {
  error: {
    // 请求级错误处理器
    handler: (error) => {
      ElMessage.warning('自定义错误：' + error.message);
      return Promise.reject(error);
    },
    // 请求级重试配置
    retry: {
      enabled: true,
      maxRetries: 2,
      delay: 500
    }
  }
});

# 配置项

参数 说明 类型 默认值
baseURL 请求基础URL string ''
timeout 超时时间(ms) number 10000
withCredentials 跨域请求时是否携带凭证 boolean true
headers 请求头配置 object -
loading loading配置 LoadingOptions -
cache 缓存配置 CacheOptions -
error 错误处理配置 ErrorOptions -
auth 认证配置 AuthOptions -

# Flags

参数 说明 类型 默认值
enableLoading 是否启用loading boolean true
enableEncrypt 是否启用加密 boolean false
enableDecrypt 是否启用解密 boolean true
enableCache 是否启用缓存 boolean false
cacheType 缓存类型 string 'none'
enableRetry 是否启用重试 boolean false
enableToken 是否启用token boolean true

# LoadingOptions

参数 说明 类型 默认值
enabled 是否启用loading boolean true
service 自定义loading服务 LoadingService -
delay 显示延迟时间(ms) number 0
minDuration 最小显示时间(ms) number 500

# CacheOptions

参数 说明 类型 默认值
enabled 是否启用缓存 boolean false
type 缓存类型 'temporary' | 'memory' | 'session' | 'persistent' 'none'
maxAge 缓存时间(ms) number -
exclude 排除的请求路径 RegExp[] []
excludeMethods 排除的请求方法 string[] []

# ErrorOptions

参数 说明 类型 默认值
handler 全局错误处理器 Function -
retry 重试配置 RetryOptions -

# RetryOptions

参数 说明 类型 默认值

maxRetries 最大重试次数 number 3
delay 重试延迟(ms) number 1000
retryableStatusCodes 可重试的状态码 number[] [408,500,502,503,504]
