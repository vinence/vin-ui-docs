# Tree 树形控件

基于 Element UI 的 Tree 组件进行二次封装，用于对比两个版本间树节点的选中状态变化，特别适用于省市区等行政区划数据的展示和对比。

# 功能特性

  • 智能层级处理 - 自动识别省市区层级关系
  • 状态对比 - 清晰展示节点选中状态的变更
  • 关联选择 - 支持父子节点关联选择
  • 搜索过滤 - 支持按名称和编码搜索
  • 变更汇总 - 直观展示所有变更的节点路径

# 基础用法

通过 tree-data 传入完整的树形数据，通过 v-model 和 previous-checked-keys 进行新旧版本对比。

变更信息
当前选中 330102(区), 320104(区)
上一版本选中 330102(区)
变更节点

基础示例展示了如何对比两个版本间节点的选中状态变化。组件会自动处理父子节点的关联关系。

<template>
  <div class="example-tree">
    <vin-diff-tree
      :tree-data="treeData"
      v-model="currentChecked"
      :previous-checked-keys="previousChecked"
      @check="handleCheck"
    />
    
    <div class="example-info">
      <el-descriptions title="变更信息" :column="1" border>
        <el-descriptions-item label="当前选中">
          {{ getSelectedInfo }}
        </el-descriptions-item>
        <el-descriptions-item label="上一版本选中">
          {{ getPreviousSelectedInfo }}
        </el-descriptions-item>
        <el-descriptions-item label="变更节点">
          {{ diffNodes.join(', ') }}
        </el-descriptions-item>
      </el-descriptions>
    </div>
  </div>
</template>

<script>
export default {
  name: 'DiffTreeBasic',

  computed: {
    // 获取当前选中信息
    getSelectedInfo() {
      return this.currentChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    },

    // 获取上一版本选中信息
    getPreviousSelectedInfo() {
      return this.previousChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    }
  },

  methods: {
    // 获取节点层级
    getNodeLevel(id) {
      const code = id.toString();
      if (code.endsWith('0000')) return 1;
      if (code.endsWith('00')) return 2;
      return 3;
    },

    handleCheck({ checkedKeys, diffInfo }) {
      // 获取变更的节点
      this.diffNodes = Object.entries(diffInfo)
        .filter(([, type]) => type === 'add' || type === 'remove')
        .map(([key]) => key);
    }
  },

  data() {
    return {
      diffNodes: [],
      // 完整的省市区树形数据
      treeData: [
        {
          id: '330000',
          name: '浙江省',
          children: [
            {
              id: '330100',
              name: '杭州市',
              children: [
                { id: '330102', name: '上城区' },
                { id: '330103', name: '下城区' },
                { id: '330104', name: '江干区' }
              ]
            },
            {
              id: '330200',
              name: '宁波市',
              children: [
                { id: '330203', name: '海曙区' },
                { id: '330205', name: '江北区' }
              ]
            }
          ]
        },
        {
          id: '320000',
          name: '江苏省',
          children: [
            {
              id: '320100',
              name: '南京市',
              children: [
                { id: '320102', name: '玄武区' },
                { id: '320104', name: '秦淮区' },
                { id: '320105', name: '建邺区' }
              ]
            }
          ]
        }
      ],
      currentChecked: ['330102', '320104'],    // 当前选中：浙江上城区、江苏秦淮区
      previousChecked: ['330102'],             // 之前选中：浙江上城区
    }
  }
}
</script>

# 不同层级数据

支持处理不同层级的数据，包括省级、市级或混合层级的数据。

数据信息
数据类型 市级数据
当前选中 330100(市), 320100(市)
上一版本选中 330100(市)
变更信息

组件会根据节点ID自动识别层级关系，确保数据展示的完整性：

  • 选中省级节点时，自动关联下级市区
  • 选中市级节点时，自动关联所属省份
  • 支持混合层级数据的处理
<template>
  <div class="example-tree">
    <div class="example-controls">
      <el-radio-group v-model="dataType" size="small">
        <el-radio-button label="province">省级数据</el-radio-button>
        <el-radio-button label="city">市级数据</el-radio-button>
        <el-radio-button label="mixed">混合数据</el-radio-button>
      </el-radio-group>
    </div>

    <vin-diff-tree
      :tree-data="fullTreeData"
      v-model="currentChecked"
      :previous-checked-keys="previousChecked"
      :show-summary="true"
      @check="handleCheck"
    />

    <div class="example-info">
      <el-descriptions title="数据信息" :column="1" border>
        <el-descriptions-item label="数据类型">
          {{ dataTypeLabel }}
        </el-descriptions-item>
        <el-descriptions-item label="当前选中">
          {{ getSelectedInfo }}
        </el-descriptions-item>
        <el-descriptions-item label="上一版本选中">
          {{ getPreviousSelectedInfo }}
        </el-descriptions-item>
        <el-descriptions-item label="变更信息">
          <div v-for="(info, type) in diffInfo" :key="type">
            {{ type }}: {{ info.join(', ') }}
          </div>
        </el-descriptions-item>
      </el-descriptions>
    </div>
  </div>
</template>

<script>
export default {
  name: 'DiffTreeMixed',
  data() {
    return {
      dataType: 'city',
      diffInfo: {},
      // 完整的树形数据
      fullTreeData: [
        {
          id: '330000',
          name: '浙江省',
          children: [
            {
              id: '330100',
              name: '杭州市',
              children: [
                { id: '330102', name: '上城区' },
                { id: '330103', name: '下城区' }
              ]
            },
            {
              id: '330200',
              name: '宁波市',
              children: [
                { id: '330203', name: '海曙区' },
                { id: '330205', name: '江北区' }
              ]
            }
          ]
        },
        {
          id: '320000',
          name: '江苏省',
          children: [
            {
              id: '320100',
              name: '南京市',
              children: [
                { id: '320102', name: '玄武区' },
                { id: '320104', name: '秦淮区' }
              ]
            }
          ]
        }
      ],
      // 不同类型的选中数据
      checkedData: {
        province: {
          current: ['330000'],              // 浙江省
          previous: ['320000']              // 江苏省
        },
        city: {
          current: ['330100', '320100'],    // 杭州市、南京市
          previous: ['330100']              // 杭州市
        },
        mixed: {
          current: ['330000', '320100'],    // 浙江省、南京市
          previous: ['330102', '320000']    // 上城区、江苏省
        }
      }
    }
  },
  computed: {
    currentChecked: {
      get() {
        return this.checkedData[this.dataType].current;
      },
      set(value) {
        this.checkedData[this.dataType].current = value;
      }
    },
    previousChecked() {
      return this.checkedData[this.dataType].previous;
    },
    dataTypeLabel() {
      const labels = {
        province: '省级数据',
        city: '市级数据',
        mixed: '混合层级数据'
      };
      return labels[this.dataType];
    },
    getSelectedInfo() {
      return this.currentChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    },
    getPreviousSelectedInfo() {
      return this.previousChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    }
  },
  methods: {
    handleCheck({ checkedKeys, diffInfo }) {
      // 处理变更信息
      const formatted = {};
      Object.entries(diffInfo).forEach(([key, type]) => {
        const level = this.getNodeLevel(key);
        const levelLabel = ['省级', '市级', '区级'][level - 1];
        formatted[levelLabel] = formatted[levelLabel] || [];
        formatted[levelLabel].push(key);
      });
      this.diffInfo = formatted;
    },
    getNodeLevel(id) {
      if (id.endsWith('0000')) return 1;
      if (id.endsWith('00')) return 2;
      return 3;
    }
  }
}
</script>

# 搜索功能

提供灵活的搜索和过滤功能。

搜索信息
搜索模式 按名称搜索
匹配节点 无匹配节点

支持多种搜索方式：

  • 按节点名称搜索
  • 按节点编码搜索
  • 实时过滤显示结果
<template>
  <div class="example-tree">
    <div class="example-controls">
      <el-input
        v-model="keyword"
        placeholder="输入关键字过滤节点"
        clearable
        style="width: 200px"
      >
        <template #prefix>
          <i class="el-icon-search"></i>
        </template>
      </el-input>

      <el-radio-group v-model="searchMode" size="small">
        <el-radio-button label="name">按名称</el-radio-button>
        <el-radio-button label="id">按编码</el-radio-button>
      </el-radio-group>
    </div>

    <vin-diff-tree
      :tree-data="treeData"
      v-model="currentChecked"
      :previous-checked-keys="previousChecked"
      :show-toolbar="false"
      :show-search="true"
      :default-expand-all="false"
      :filter-node-method="filterNode"
      @check="handleCheck"
    />

    <div class="example-info">
      <el-descriptions title="搜索信息" :column="1" border>
        <el-descriptions-item label="搜索模式">
          {{ searchMode === 'name' ? '按名称搜索' : '按编码搜索' }}
        </el-descriptions-item>
        <el-descriptions-item label="匹配节点">
          <template v-if="matchedNodes.length">
            <el-tag 
              v-for="node in matchedNodes" 
              :key="node.id"
              size="small"
              style="margin-right: 4px"
            >
              {{ node.name }} ({{ node.id }})
            </el-tag>
          </template>
          <template v-else>
            无匹配节点
          </template>
        </el-descriptions-item>
      </el-descriptions>
    </div>
  </div>
</template>

<script>
export default {
  name: 'DiffTreeSearch',
  data() {
    return {
      keyword: '',
      searchMode: 'name',
      matchedNodes: [],
      treeData: [
        {
          id: '330000',
          name: '浙江省',
          children: [
            {
              id: '330100',
              name: '杭州市',
              children: [
                { id: '330102', name: '上城区' },
                { id: '330103', name: '下城区' },
                { id: '330104', name: '江干区' }
              ]
            },
            {
              id: '330200',
              name: '宁波市',
              children: [
                { id: '330203', name: '海曙区' },
                { id: '330205', name: '江北区' }
              ]
            }
          ]
        },
        {
          id: '320000',
          name: '江苏省',
          children: [
            {
              id: '320100',
              name: '南京市',
              children: [
                { id: '320102', name: '玄武区' },
                { id: '320104', name: '秦淮区' },
                { id: '320105', name: '建邺区' }
              ]
            }
          ]
        }
      ],
      currentChecked: [],
      previousChecked: []
    }
  },
  methods: {
    filterNode(value, data) {
      if (!value) return true;
      
      const searchValue = value.toLowerCase();
      if (this.searchMode === 'name') {
        return data.name.toLowerCase().includes(searchValue);
      } else {
        return data.id.includes(searchValue);
      }
    },
    handleCheck({ checkedKeys, checkedNodes }) {
      // 更新匹配节点列表
      if (this.keyword) {
        this.matchedNodes = checkedNodes.filter(node => 
          this.filterNode(this.keyword, node)
        );
      } else {
        this.matchedNodes = [];
      }
    }
  },
  watch: {
    keyword(val) {
      this.$nextTick(() => {
        const tree = this.$refs.tree;
        if (tree) {
          tree.filter(val);
        }
      });
    }
  }
}
</script>

export default {

    getSelectedInfo() {
      return this.currentChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    },

    // 获取上一版本选中信息
    getPreviousSelectedInfo() {
      return this.previousChecked.map(id => {
        const level = this.getNodeLevel(id);
        const levelLabel = ['省', '市', '区'][level - 1];
        return `${id}(${levelLabel})`;
      }).join(', ');
    }
  },

  methods: {
    // 获取节点层级
    getNodeLevel(id) {
      const code = id.toString();
      if (code.endsWith('0000')) return 1;
      if (code.endsWith('00')) return 2;

  data() {
    return {
      showToolbar: true,

      showSummary: true,
      diffNodes: [],

      treeData: [

          id: '330000',
          name: '浙江省',
          children: [

# API

# Props

参数 说明 类型 可选值 默认值
tree-data 树形数据 array - -
current-checked-keys 当前版本选中的节点key数组 array - -
previous-checked-keys 上个版本选中的节点key数组 array - -
props 配置选项 object - -
show-toolbar 是否显示工具栏 boolean - true
show-search 是否显示搜索框 boolean - true
show-summary 是否显示变更汇总 boolean - true
default-expand-all 是否默认展开所有节点 boolean - true

# props

参数 说明 类型 可选值 默认值
children 子节点 key string - children
label 节点标签 key string - name
id 节点唯一标识 key string - id

# Events

事件名称 说明 回调参数
check 节点选中状态变化时的回调 { checkedKeys, checkedNodes, diffInfo }

# Methods

方法名 说明 参数
getCheckedNodes 获取当前选中的节点数据 -
getCheckedKeys 获取当前选中的节点key -
setCheckedKeys 设置选中的节点key (keys)
filter 过滤节点 (text)
