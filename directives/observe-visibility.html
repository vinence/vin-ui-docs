# 元素可见性监听指令

# 基础用法

使用 v-observe-visibility 指令可以监听元素进入或离开视口的事件。

向下滚动查看效果
观察元素 
<template>
  <div class="visibility-demo">
    <div class="scroll-container">
      <div class="placeholder">向下滚动查看效果</div>
      <div 
        class="observe-box"
        v-observe-visibility="handleVisibilityChange"
      >
        观察元素
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      timer: null
    }
  },
  methods: {
    handleVisibilityChange(isVisible) {
      // 使用 Element UI 的消息提示
      this.$message({
        message: isVisible ? '元素进入视图！' : '元素离开视图！',
        type: isVisible ? 'success' : 'warning',
        duration: 2000
      })
    }
  }
}
</script>

# 配置选项

可以通过配置对象设置触发阈值、单次触发等选项。当元素显示 50% 时触发回调。

向下滚动，当元素显示 50% 时触发
观察元素 (显示50%触发) 
<template>
  <div class="visibility-demo">
    <div class="scroll-container">
      <div class="placeholder">向下滚动，当元素显示 50% 时触发</div>
      <div 
        class="observe-box"
        v-observe-visibility="{
          callback: handleVisibilityChange,
          intersection: { threshold: 0.5 },
          once: true
        }"
      >
        观察元素 (显示50%触发)
      </div>
    </div>
  </div>
</template>

<script>
export default {
  methods: {
    handleVisibilityChange(isVisible, entry) {
      if (isVisible) {
        this.$message({
          message: `元素已显示 ${Math.round(entry.intersectionRatio * 100)}%`,
          type: 'success',
          duration: 2000
        })
      }
    }
  }
}
</script>

# 节流控制

使用节流选项控制回调触发频率，避免频繁触发。

快速滚动，消息每秒最多触发一次
观察元素 (节流1秒)
触发次数: 0
<template>
  <div class="visibility-demo">
    <div class="scroll-container">
      <div class="placeholder">快速滚动，消息每秒最多触发一次</div>
      <div 
        class="observe-box"
        v-observe-visibility="{
          callback: handleVisibilityChange,
          throttle: 1000,
          throttleOptions: { leading: 'visible' }
        }"
      >
        观察元素 (节流1秒)
        <div class="counter">触发次数: {{ count }}</div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      count: 0
    }
  },
  methods: {
    handleVisibilityChange(isVisible) {
      this.count++
      this.$message({
        message: `元素${isVisible ? '可见' : '不可见'} (第 ${this.count} 次触发)`,
        type: 'info',
        duration: 1000
      })
    }
  }
}
</script>

# 图片懒加载

结合可见性监听实现图片懒加载，提升页面性能。

<template>
  <div class="lazy-demo">
    <div class="image-list">
      <div 
        v-for="(image, index) in images" 
        :key="index"
        class="image-item"
        v-observe-visibility="{
          callback: (visible) => handleImageVisible(visible, index),
          once: true
        }"
      >
        <img 
          :src="image.loaded ? image.url : placeholder" 
          :alt="'Image ' + (index + 1)"
        />
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      placeholder: 'data:image/svg+xml;base64,PHN2ZyB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciIHZpZXdCb3g9IjAgMCAzMjAgMjAwIj48cmVjdCB3aWR0aD0iMzIwIiBoZWlnaHQ9IjIwMCIgZmlsbD0iI2YwZjBmMCIvPjx0ZXh0IHg9IjUwJSIgeT0iNTAlIiBkb21pbmFudC1iYXNlbGluZT0ibWlkZGxlIiB0ZXh0LWFuY2hvcj0ibWlkZGxlIiBmb250LWZhbWlseT0ic2Fucy1zZXJpZiIgZm9udC1zaXplPSIyNiIgZmlsbD0iIzk5OSI+TG9hZGluZy4uLjwvdGV4dD48L3N2Zz4=',
      images: Array.from({ length: 10 }, (_, i) => ({
        url: `https://picsum.photos/320/200?random=${i}`,
        loaded: false
      }))
    }
  },
  methods: {
    handleImageVisible(visible, index) {
      if (visible) {
        this.images[index].loaded = true
      }
    }
  }
}
</script>

# 无限滚动

监听底部元素可见性来实现无限滚动加载。

加载中...
<template>
  <div class="infinite-demo">
    <div class="list-container">
      <div 
        v-for="item in items" 
        :key="item.id"
        class="list-item"
      >
        {{ item.content }}
      </div>
      <div 
        v-if="!isFinished"
        class="loading-trigger"
        v-observe-visibility="{
          callback: handleLoadMore,
          throttle: 300
        }"
      >
        <div v-if="loading" class="loading-text">加载中...</div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      items: [],
      page: 1,
      loading: false,
      isFinished: false
    }
  },
  created() {
    this.loadData()
  },
  methods: {
    async loadData() {
      this.loading = true
      // 模拟异步加载
      await new Promise(resolve => setTimeout(resolve, 1000))
      
      const newItems = Array.from({ length: 10 }, (_, i) => ({
        id: (this.page - 1) * 10 + i,
        content: `列表项 ${(this.page - 1) * 10 + i + 1}`
      }))
      
      this.items.push(...newItems)
      this.loading = false
      
      if (this.page >= 5) {
        this.isFinished = true
      }
    },
    handleLoadMore(visible) {
      if (visible && !this.loading && !this.isFinished) {
        this.page++
        this.loadData()
      }
    }
  }
}
</script>

}

}

}
.loading-trigger {
  padding: 20px;

}

# 数据统计

监听广告位或重要内容的曝光数据。

广告位展示区域
广告位曝光时间：-
曝光比例：0%
<template>
  <div class="stats-demo">
    <div class="content-container">
      <div class="banner-section">
        <div 
          class="ad-banner"
          v-observe-visibility="{
            callback: handleBannerExposure,
            intersection: { threshold: 0.5 },
            once: true
          }"
        >
          广告位展示区域
          <div class="exposure-tag" v-if="bannerExposed">
            已曝光
          </div>
        </div>
      </div>
      
      <div class="stats-info">
        <div>广告位曝光时间：{{ exposureTime || '-' }}</div>
        <div>曝光比例：{{ exposureRatio }}%</div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      bannerExposed: false,
      exposureTime: '',
      exposureRatio: 0
    }
  },
  methods: {
    handleBannerExposure(isVisible, entry) {
      if (isVisible) {
        this.bannerExposed = true
        this.exposureTime = new Date().toLocaleTimeString()
        this.exposureRatio = Math.round(entry.intersectionRatio * 100)
      }
    }
  }
}
</script>

# 指令参数

# 简单用法

参数 说明 类型 默认值
value 可见性变化回调函数 Function(isVisible: boolean, entry: IntersectionObserverEntry) -

# 配置对象

参数 说明 类型 默认值
callback 可见性变化回调函数 Function(isVisible: boolean, entry: IntersectionObserverEntry) -
once 是否只触发一次 boolean false
throttle 节流时间（毫秒） number -
intersection IntersectionObserver 配置项 object -
throttleOptions 节流选项 object -

# intersection 配置项

参数 说明 类型 默认值
threshold 触发回调的可见比例阈值 number 0
root 指定根(root)元素 Element null
rootMargin 根元素的外边距 string '0px'

# throttleOptions 配置项

参数 说明 类型 可选值 默认值
leading 节流首次执行时机 string 'both' / 'visible' / 'hidden' 'both'
