# 对话框拖拽指令

# 基础用法

将 v-drag-dialog 指令添加到 Dialog 组件上即可实现拖拽功能。

可拖拽对话框
<template>
  <div class="example-dialog">
    <el-button type="primary" @click="dialogVisible = true">
      打开可拖拽对话框
    </el-button>

    <el-dialog
      title="可拖拽对话框"
      :visible.sync="dialogVisible"
      width="30%"
      v-drag-dialog
    >
      <span>这是一个可以拖拽的对话框</span>
      <span slot="footer" class="dialog-footer">
        <el-button @click="dialogVisible = false">取 消</el-button>
        <el-button type="primary" @click="dialogVisible = false">确 定</el-button>
      </span>
    </el-dialog>
  </div>
</template>

<script>
export default {
  data() {
    return {
      dialogVisible: false
    }
  }
}
</script>

<style scoped>
.example-dialog {
  padding: 20px;
}
</style>

# 调整大小

添加 resize 修饰符可以启用大小调整功能。

可调整大小的对话框
<template>
  <div class="example-dialog">
    <el-button type="primary" @click="dialogVisible = true">
      打开可调整大小的对话框
    </el-button>

    <el-dialog
      title="可调整大小的对话框"
      :visible.sync="dialogVisible"
      width="30%"
      v-drag-dialog.resize
    >
      <span>这个对话框可以拖拽和调整大小</span>
      <span slot="footer" class="dialog-footer">
        <el-button @click="dialogVisible = false">取 消</el-button>
        <el-button type="primary" @click="dialogVisible = false">确 定</el-button>
      </span>
    </el-dialog>
  </div>
</template>

<script>
export default {
  data() {
    return {
      dialogVisible: false
    }
  }
}
</script>

<style scoped>
.example-dialog {
  padding: 20px;
}
</style>
